```sql
customer table
---------
CREATE TABLE CUSTOMER (
    CUSTOMERID INTEGER NOT NULL PRIMARY KEY,
    FName VARCHAR(32) NOT NULL,
    LName VARCHAR(60),
    PhoneNr VARCHAR(15)
);
```

```sql
EMPLOYEE table
---------
CREATE TABLE EMPLOYEE (
    EMPLOYEEID INTEGER NOT NULL PRIMARY KEY,
    FName VARCHAR(32) NOT NULL,
    LName VARCHAR(60),
    PhoneNr VARCHAR(15),
    JobTitle VARCHAR(20)
);
```

```sql
AMBULANCE table
---------
CREATE TABLE AMBULANCE (
    PlateNr VARCHAR(16) PRIMARY KEY,
    Type VARCHAR(16)
);
```

```sql
RESERVATION table
---------
CREATE TABLE RESERVATION (
    RESERVATIONID INTEGER NOT NULL PRIMARY KEY,
    CUSTOMERID INTEGER,
    Date  DATE,
     StartTime TIME,
    EndTime TIME,
    PlateNr VARCHAR(16),

    FOREIGN KEY (CUSTOMERID) REFERENCES CUSTOMER(CUSTOMERID) ON UPDATE CASCADE ON DELETE CASCADE,
    FOREIGN KEY (PlateNr) REFERENCES AMBULANCE(PlateNr) ON UPDATE CASCADE ON DELETE CASCADE
);
```

```sql
CREW_RIDE table
---------
CREATE TABLE CREW_RIDE (
    RESERVATIONID INTEGER,
    EMPLOYEEID INTEGER,
    PRIMARY KEY (EMPLOYEEID, RESERVATIONID),
    FOREIGN KEY (EMPLOYEEID) REFERENCES EMPLOYEE(EMPLOYEEID) ON UPDATE CASCADE ON DELETE CASCADE,
    FOREIGN KEY (RESERVATIONID) REFERENCES RESERVATION(RESERVATIONID) ON UPDATE CASCADE ON DELETE CASCADE
);
```

    EMPLOYEEID INTEGER NOT NULL PRIMARY KEY,
    FName VARCHAR(32) NOT NULL,
    LName VARCHAR(60),
    PhoneNr VARCHAR(15),
    JobTitle VARCHAR(20)
);

CREATE TABLE AMBULANCE (
    PlateNr VARCHAR(16) PRIMARY KEY,

);

CREATE TABLE RESERVATION (
    RESERVATIONID INTEGER NOT NULL PRIMARY KEY,
    CUSTOMERID INTEGER,

    EMPLOYEEID INTEGER,
    PRIMARY KEY (EMPLOYEEID, RESERVATIONID),
